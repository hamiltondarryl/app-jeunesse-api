              contains: zone, 
              mode: 'insensitive'

              nom: { 
                contains: zone, 
                mode: 'insensitive' 
              } 
            },
            include: { departements: true }
          });

            console.warn(`⚠️ Département non trouvé pour la zone: "${zone}", utilisé le premier département de: "${province.nom}"`);
          } else {
            // Fallback: utiliser le premier département disponible

            if (fallbackDepartement) {
              departementConnect.push({ id: fallbackDepartement.id });
              console.warn(`⚠️ Zone non trouvée: "${zone}", utilisé département fallback`);
            }
          }
        }

      .filter((d) => d.length > 0);

          connectOrCreate: domaineConnections,
